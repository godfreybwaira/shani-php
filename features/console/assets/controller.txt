<?php

namespace {namespace} {

    use features\ds\map\ReadMap;
    use gui\WebUIBuilder;
    use shani\http\HttpResponse;
    use shani\launcher\App;

    final class {controller_name}
    {

        private readonly App $app;

        public function __construct(App $app)
        {
            $this->app = $app;
        }

        public function {fn_name}(): HttpResponse
        {
            $data = new ReadMap([
                'name' => get_current_user(),
                'title' => 'developer'
            ]);

            $builder = new WebUIBuilder($data);
            $builder->title('{module_name}')
            ->description('Description of {module_name}')
            ->attr->addOne('name', $data->getOne('name'));

            return HttpResponse::withBody($builder);
        }
    }

}
